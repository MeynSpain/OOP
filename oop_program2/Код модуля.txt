#include <iostream>
#include <cstdlib>  // Для рандомных чисел
namespace Kib       // Пространство имён Kib
{

    int* Filling(int mas[], int di, int l) {         // Функция заполнения массива...
        srand(static_cast<unsigned int>(time(0)));   // Псевдослучайность
        for (int i = 0; i < l; i++)
            mas[i] = rand() % (di + 1);              // ...случайными числами
        return mas;
    }

    void Output(int mas[], int di, int l) {         // Процедура вывода массива
        using std::cout;                            // using-стейтменты
        using std::endl;
        for (int i = 0; i < l; i++)
            cout << "mas[" << i << "]=" << mas[i] << endl;
    }
}
